## K-fold (4 folds)
folds <- vfold_cv(housing_df, v = 4, strata = NULL, repeats = 1)
workflow <- workflow() %>%
  add_recipe(rec) %>%
  add_model(lin_reg)


results <- workflow %>%
  fit_resamples(resamples = folds, 
                metrics = metric_set(rmse, rsq))
collect_metrics(results)

# Specify a different model
lasso_mod <- linear_reg(penalty = 0.001, mixture = 1) %>%

  add_recipe(rec) %>%
  add_model(lasso_mod)
